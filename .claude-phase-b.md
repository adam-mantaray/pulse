```typescript
// Send task to agent for flesh-out or execution via OpenClaw sessions_send
export const sendToAgent = action({
  args: {
    taskId: v.id("haradaTasks"),
    agentName: v.string(),       // "tarek" | "rami" | "nadia" | "adam" | "omar"
    assignmentType: v.union(v.literal("flesh_out"), v.literal("execute")),
    openclaw_session_key: v.optional(v.string()), // agent's OpenClaw session key
  },
  handler: async (ctx, args) => {
    const task = await ctx.runQuery(api.haradaTasks.getById, { taskId: args.taskId });
    if (!task) throw new Error("Task not found");
    const chart = await ctx.runQuery(api.harada.get, { chartId: task.chartId });
    if (!chart) throw new Error("Chart not found");

    // Build the structured message for the agent
    const subGoalName = chart.subGoals[task.subGoalIndex] || `Sub-Goal ${task.subGoalIndex + 1}`;
    const actionText = chart.actions[task.subGoalIndex]?.[task.actionIndex] || `Action ${task.actionIndex + 1}`;

    let message: string;
    if (args.assignmentType === "flesh_out") {
      message = `📋 **Task for flesh-out:**\n\n**Vision:** ${chart.mainGoal}\n**Sub-Goal:** ${subGoalName}\n**Action:** ${actionText}\n**Task:** ${task.title}\n${task.notes ? `**Context:** ${task.notes}\n` : ""}\nPlease write a detailed plan/breakdown for this task. When done, call the Pulse API:\nPOST ${process.env.CONVEX_SITE_URL}/pulse/tasks/${task._id}/flesh-out\nBody: { "plan": "your markdown plan", "agentName": "${args.agentName}" }\nHeader: X-Pulse-Secret: ${process.env.PULSE_AGENT_SECRET}`;
    } else {
      message = `⚡ **Task to execute:**\n\n**Vision:** ${chart.mainGoal}\n**Sub-Goal:** ${subGoalName}\n**Action:** ${actionText}\n**Task:** ${task.title}\n${task.fleshOutPlan ? `**Approved Plan:**\n${task.fleshOutPlan}\n` : ""}${task.notes ? `**Context:** ${task.notes}\n` : ""}\nExecute this task. When done, call:\nPOST ${process.env.CONVEX_SITE_URL}/pulse/tasks/${task._id}/complete\nBody: { "notes": "completion summary", "agentName": "${args.agentName}" }\nHeader: X-Pulse-Secret: ${process.env.PULSE_AGENT_SECRET}`;
    }

    // Update task status
    await ctx.runMutation(api.haradaTasks.update, {
      taskId: args.taskId,
      assignedAgentName: args.agentName,
      status: args.assignmentType === "flesh_out" ? "fleshing_out" : "executing",
    });

    // Send message to agent via agentMessages (stores in Convex)
    // Agent will poll/receive via their OpenClaw session
    await ctx.runMutation(api.agentMessages.sendToAgent, {
      agentName: args.agentName,
      content: message,
      relatedTaskId: args.taskId,
    });

    return { sent: true };
  },
});

// Called by agent HTTP endpoint when flesh-out plan is ready
export const receiveFleshOut = mutation({
  args: {
    taskId: v.id("haradaTasks"),
    plan: v.string(),
    agentName: v.string(),
  },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.taskId, {
      fleshOutPlan: args.plan,
      fleshOutAt: Date.now(),
      status: "pending_review",
      updatedAt: Date.now(),
    });
    // TODO: trigger push notification (expo-notifications)
    return { status: "pending_review" };
  },
});

// Ahmed approves the plan — sends execute instruction
export const approve = mutation({
  args: { taskId: v.id("haradaTasks") },
  handler: async (ctx, args) => {
    const task = await ctx.db.get(args.taskId);
    if (!task) throw new Error("Task not found");
    await ctx.db.patch(args.taskId, {
      approvedAt: Date.now(),
      status: "approved",
      updatedAt: Date.now(),
    });
    return { status: "approved" };
  },
});

// Called by agent when execution is complete
export const receiveComplete = mutation({
  args: {
    taskId: v.id("haradaTasks"),
    notes: v.string(),
    agentName: v.string(),
  },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.taskId, {
      executionNotes: args.notes,
      completedAt: Date.now(),
      status: "done",
      updatedAt: Date.now(),
    });
    // Check if all tasks under this action are done → auto-mark action done
    const task = await ctx.db.get(args.taskId);
    if (task) {
      const siblings = await ctx.db
        .query("haradaTasks")
        .withIndex("by_chart_action", (q) =>
          q.eq("chartId", task.chartId)
           .eq("subGoalIndex", task.subGoalIndex)
           .eq("actionIndex", task.actionIndex)
        )
        .collect();
      const allDone = siblings.every((t) => t.status === "done" || t._id === args.taskId);
      if (allDone) {
        await ctx.runMutation(api.harada.autoCompleteAction, {
          chartId: task.chartId,
          subGoalIndex: task.subGoalIndex,
          actionIndex: task.actionIndex,
        });
      }
    }
    return { status: "done" };
  },
});

// Query for pending review tasks (for badge count)
export const listPendingReview = query({
  args: { userId: v.id("users") },
  handler: async (ctx, args) => {
    return await ctx.db
      .query("haradaTasks")
      .withIndex("by_user", (q) => q.eq("userId", args.userId))
      .filter((q) => q.eq(q.field("status"), "pending_review"))
      .collect();
  },
});

export const getById = query({
  args: { taskId: v.id("haradaTasks") },
  handler: async (ctx, args) => ctx.db.get(args.taskId),
});
```

### 2. Update `convex/agentMessages.ts`

Add `sendToAgent` mutation (internal — writes outbound message):
```typescript
export const sendToAgent = mutation({
  args: {
    agentName: v.string(),
    content: v.string(),
    relatedTaskId: v.optional(v.id("haradaTasks")),
  },
  handler: async (ctx, args) => {
    // Find userId — use first user (single-user app for now)
    const users = await ctx.db.query("users").take(1);
    const userId = users[0]?._id;
    if (!userId) throw new Error("No user found");
    return await ctx.db.insert("agentMessages", {
      userId,
      agentName: args.agentName,
      direction: "outbound",
      content: args.content,
      timestamp: Date.now(),
      read: true,
      delivered: false,
      relatedTaskId: args.relatedTaskId,
    });
  },
});
```

### 3. Update `convex/http.ts`

```typescript
// POST /pulse/tasks/:taskId/flesh-out
http.route({
  path: "/pulse/tasks/:taskId/flesh-out", // note: convex doesn't support path params — use query or body
  // Actually use: POST /pulse/tasks/flesh-out  with taskId in body
  method: "POST",
  handler: httpAction(async (ctx, request) => {
    // Verify X-Pulse-Secret header
    const secret = request.headers.get("X-Pulse-Secret");
    if (secret !== process.env.PULSE_AGENT_SECRET) {
      return new Response("Unauthorized", { status: 401 });
    }
    const body = await request.json();
    const { taskId, plan, agentName } = body;
    await ctx.runMutation(api.haradaTasks.receiveFleshOut, { taskId, plan, agentName });
    return new Response(JSON.stringify({ ok: true }), { status: 200 });
  }),
});
```
