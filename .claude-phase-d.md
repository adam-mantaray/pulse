### 1. `convex/dashboard.ts` — single aggregated query

```typescript
export const getSummary = query({
  args: { userId: v.id("users") },
  handler: async (ctx, args) => {
    // 1. Active Harada chart stats
    const activeChart = await ctx.db.query("haradaCharts")
      .withIndex("by_user_active", q => q.eq("userId", args.userId).eq("isActive", true))
      .first();
    
    let harada = { activeChart: null, actionsFilledCount: 0, actionsDoneCount: 0, pendingReviewCount: 0, totalActions: 64 };
    if (activeChart) {
      const filledCount = activeChart.actions.flat().filter(a => a.length > 0).length;
      const doneCount = (activeChart.actionsDone ?? []).flat().filter(Boolean).length;
      const pendingTasks = await ctx.db.query("haradaTasks")
        .withIndex("by_user", q => q.eq("userId", args.userId))
        .filter(q => q.eq(q.field("status"), "pending_review"))
        .collect();
      harada = {
        activeChart: { id: activeChart._id, title: activeChart.title, mainGoal: activeChart.mainGoal },
        actionsFilledCount: filledCount,
        actionsDoneCount: doneCount,
        pendingReviewCount: pendingTasks.length,
        totalActions: 64,
      };
    }

    // 2. Habits for today
    const today = new Date().toISOString().split("T")[0];
    const habits = await ctx.db.query("habits")
      .withIndex("by_user", q => q.eq("userId", args.userId))
      .filter(q => q.eq(q.field("isActive"), true))
      .collect();
    const completions = await Promise.all(habits.map(h =>
      ctx.db.query("habitCompletions")
        .withIndex("by_habit_date", q => q.eq("habitId", h._id).eq("date", today))
        .first()
    ));
    // Resolve sub-goal labels
    const habitsWithContext = await Promise.all(habits.map(async (h, i) => {
      let subGoalLabel: string | null = null;
      if (h.haradaChartId && h.haradaSubGoalIndex !== undefined) {
        const chart = await ctx.db.get(h.haradaChartId);
        subGoalLabel = chart?.subGoals[h.haradaSubGoalIndex] ?? null;
      }
      return { id: h._id, name: h.name, emoji: h.emoji, streak: h.currentStreak, completedToday: !!completions[i], subGoalLabel };
    }));

    // 3. OKR summary
    const now = new Date();
    const quarter = `Q${Math.ceil((now.getMonth() + 1) / 3)}-${now.getFullYear()}`;
    const objectives = await ctx.db.query("objectives")
      .withIndex("by_user_quarter", q => q.eq("userId", args.userId).eq("quarter", quarter))
      .collect();
    const objsWithKRs = await Promise.all(objectives.map(async (obj) => {
      const krs = await ctx.db.query("keyResults").withIndex("by_objective", q => q.eq("objectiveId", obj._id)).collect();
      const progress = krs.length > 0 ? Math.round(krs.reduce((s, kr) => s + kr.progress, 0) / krs.length) : 0;
      return { id: obj._id, title: obj.title, progress };
    }));
    const overallProgress = objsWithKRs.length > 0
      ? Math.round(objsWithKRs.reduce((s, o) => s + o.progress, 0) / objsWithKRs.length)
      : 0;

    // 4. Focus (active pomodoro)
    const ninetyMinsAgo = Date.now() - 90 * 60 * 1000;
    const todaySessions = await ctx.db.query("pomodoroSessions")
      .withIndex("by_user", q => q.eq("userId", args.userId))
      .filter(q => q.gte(q.field("startedAt"), ninetyMinsAgo - 86400000))
      .collect();
    const todayStart = new Date(); todayStart.setHours(0,0,0,0);
    const todaySessionsFiltered = todaySessions.filter(s => s.startedAt >= todayStart.getTime());
    const activeSession = todaySessions.find(s => !s.completedAt && s.startedAt >= ninetyMinsAgo) ?? null;
    const totalMinutes = todaySessionsFiltered.reduce((s, sess) => s + (sess.actualDurationMinutes ?? sess.plannedDurationMinutes), 0);

    // 5. Agents
    const agents = await ctx.db.query("agents").collect();

    // 6. Blockers from linearCache
    const linearData = await ctx.db.query("linearCache").collect();
    const blockers = linearData.reduce((s, p) => s + p.blockedTasks, 0);

    return {
      harada,
      habits: { todayHabits: habitsWithContext, doneCount: completions.filter(Boolean).length, totalCount: habits.length },
      okrs: { currentQuarter: quarter, overallProgress, objectives: objsWithKRs.slice(0, 3) },
      focus: { activeSession, todaySessions: todaySessionsFiltered.length, todayMinutes: totalMinutes },
      agents: agents.map(a => ({ agentName: a.name, status: a.status, currentTask: a.currentTask })),
      blockers,
    };
  },
});
```

### 2. Rebuild `app/(tabs)/index.tsx` — Unified Home

- Two concentric SVG rings using react-native-svg:
  - Outer ring: `actionsDoneCount / 64` — color changes: 0-33% = `textTertiary`, 34-66% = `streak` (gold), 67-100% = `accent`

  - Center: mainGoal text (max 2 lines, Fraunces, small)
  - Below center: "X/64 done" in DM Sans tiny
- Animated fill on load (Reanimated withTiming)
- If no chart: single ring with "Set Vision" center text
- Tap → navigate to Vision tab

**C. Focus Card**
- If `focus.activeSession` exists: 
  - Card with live countdown (useEffect setInterval)
  - Task label + "🍅 In progress" badge
  - Tap → navigate to `/pomodoro`
- If idle:
  - Card: "🍅 What are you working on?" muted text
  - Tap → navigate to `/pomodoro`
- If today sessions > 0: `focus.todaySessions` sessions · `focus.todayMinutes` min below

**D. Today's Habits**
